country_event = { #Dakotan culture
	id = cultureflavor.1
	title = "cultureflavor.1.t"
	desc = "cultureflavor.1.d"
	picture = CULTURE_eventPicture

	fire_only_once = yes

	is_triggered_only = yes

	option = { # Unite Dakotan culture
		name = "cultureflavor.1.opta"
		if = {
			limit = {
				NOT = {
					primary_culture = us_dakotan
				}
			}
			change_primary_culture = us_dakotan
		}
		if = {
			limit = {
				primary_culture = north_dakotan
				accepted_culture = south_dakotan
			}
			add_dip_power = 100
		}
		if = {
			limit = {
				primary_culture = south_dakotan
				accepted_culture = north_dakotan
			}
			add_dip_power = 100
		}
		custom_tooltip = cultureflavor.1.opta.tt
		hidden_effect = {
			set_ruler_culture = us_dakotan
			if = {
                limit = {
                    has_heir = yes
                }
                set_heir_culture = us_dakotan
            }
            if = {
                limit = {
                    has_consort = yes
                    OR = {
                        consort_culture = south_dakotan
                        consort_culture = north_dakotan
                    }
                }
                set_consort_culture = us_dakotan
            }
			remove_accepted_culture = south_dakotan
			remove_accepted_culture = north_dakotan
			every_owned_province = {
				limit = {
					OR = {
						culture = south_dakotan
						culture = north_dakotan
					}
				}
				change_culture = us_dakotan
			}
		}
		ai_chance = {
			factor = 90
		}
	}

	option = { # Keep culture
		name = "cultureflavor.1.optb"
		add_prestige_or_monarch_power = { amount = 20}
		ai_chance = {
			factor = 10
		}
	}
}
