country_event = { # Plight of the Jeffersonians
	id = flavorUTA.3
	title = "flavorUTA.3t"
	desc = "flavorUTA.3d"
	picture = CONQUEST_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			tag = DES
			tag = UTA
			tag = SUT
		}
	}

	option = { # We have shared the plight of the Jeffersonians before. Let us aid them.
	name = flavorUTA.3.opta
	hidden_effect = {
		jefferson_area = {
			add_core = JEF
			remove_core = ROOT 
		}
		high_desert_area = {
			add_core = JEF
			remove_core = ROOT
		}
	}
	if = {
		limit = { NOT = { exists = JEF } }
		release = JEF
		hidden_effect = {
			ROOT = {
				add_historical_friend = JEF
			}
			JEF = {
				add_historical_friend = ROOT
				if = {
					limit = {
						government = republic
					}
					change_government = republic
				}
				if = {
					limit = {
						government = monarchy
					}
					change_government = monarchy
				}
				if = {
					limit = {
						government = theocracy
					}
					change_government = theocracy
				}
				else = {
					change_government = theocracy
				}
				add_government_reform = american_state_reform
				if = {
					limit = {
						ROOT = {
							religion = mormon
						}
					}
					change_religion = mormon
				}
				change_primary_culture = jeffersonian
				set_prestige_to_zero = yes
				add_adm_power = 50
				add_dip_power = 50
				add_mil_power = 50
			}
		}
	}
	if = {
		limit = { JEF = { is_subject = no } }
		if = {
			limit = { valid_for_personal_unions_trigger = yes JEF = { valid_for_personal_unions_trigger = yes } }
			create_subject = {
				who = JEF
				subject_type = personal_union
			}
		}
		else = {
			create_subject = {
				who = JEF
				subject_type = vassal
			}
		}
	}
	hidden_effect = {
		add_country_modifier = {
			name = uta_protectors_of_jefferson
			duration = -1
			hidden = yes
		}
	}
	custom_tooltip = protectors_of_jefferson_tt
	}
	
	option = { # We are better off holding this land ourselves.
	name = flavorUTA.3.optb
		add_adm_power = 50
	}
}
