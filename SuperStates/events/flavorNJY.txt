	trigger = {
		owns = 3281 # Staten Island/Richmond
	}

	option = {	# Relinquish Staten Island
		name = "flavorNJY.2opta"
		ai_chance = {
			factor = 30
				modifier = {
					factor = 1.5
					alliance_with = FROM
			}
			modifier = {
					factor = 1.1
					has_opinion = {
						who = ROOT
						value = 10
						}
			}
				modifier = {
					factor = 0.5
					is_rival = FROM
				}
			}
		FROM = {
			country_event = {
				id = flavorNJY.4
				days = 3
			}
		}
		3281 = {
			cede_province = FROM
			add_core = FROM
			remove_core = ROOT
		}
	}
	option = {	# We have rightfully won the rights to Staten Island. It is ours to keep.
		name = "flavorNJY.2optb"
		ai_chance = { factor = 70 }
		FROM = { country_event = {
		id = flavorNJY.3
		days = 3
			}
		}
	}
}
