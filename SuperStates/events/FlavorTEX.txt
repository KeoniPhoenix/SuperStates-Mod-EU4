# Adelsverein Requests Land Grants
country_event = {
	id = flavorTEX.3
	title = "flavorTEX.3t"
	desc = "flavorTEX.3d"
	picture = BIG_BOOK_eventPicture

	trigger = {
	has_country_flag = germans_accepted
	AND = {
		tag = TEX
		NOT = { primary_culture = tejano }
		NOT = { primary_culture = mexican_texan }
		}
	has_institution = new_world_i
	}
	
	mean_time_to_happen = {
		months = 60

		modifier = {
			factor = 0.50
				accepted_culture = texasdeutsch
		}
	}
	
	immediate = {
		#clr_country_flag = received_germans
		#clr_country_flag = germans_accepted
		set_country_flag = adelsverein_grants
	}
	
	option = { # Accept their request
		name = "flavorTEX.3opta"
		add_country_modifier = {
			name = "texans_displaced"
			duration = 3650
		}
		every_owned_province = {
			limit =  { 
				has_province_flag = german_development_flag
			}
			every_province_in_state = {
				limit = {
					NOT = { province_id = 2480 } # Dallas
					NOT = { province_id = 2478 } # Austin
					NOT = { province_id = 2487 } # Houston
				}
				add_base_production = 1
				add_base_manpower = 1
				add_base_tax = 1
				change_religion = protestant
				change_culture = texasdeutsch
				hidden_effect = {
					add_province_modifier = {
					name = "expelled_minority"
					duration = 5475
			}
		}
			}
		}
		add_country_modifier = { 
			name = texas_german_integration
			duration = -1  
			desc = texas_german_integration_tt
			hidden = yes
		}
		custom_tooltip = texas_german_integration_tt
		hidden_effect = {
		if = {
			limit = { 
				NOT = { accepted_culture =  texasdeutsch }
			}
				add_accepted_culture = texasdeutsch
		else = {
				add_dip_power = 100
			}
		}
	}
		add_adm_power = -25
		add_dip_power = -25
		set_country_flag = adelsverein_grants_given
	}
	option = { # Texas is for the Texans! Turn them away.
		name = "flavorTEX.3optb"
		add_prestige = 15
		add_country_modifier = {
			name = "adelsverein_refused"
			duration = 3650
		}
	}
}
