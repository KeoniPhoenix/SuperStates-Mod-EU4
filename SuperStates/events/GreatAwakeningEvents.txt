			NOT = { has_reform = papacy_reform }

			continent = north_america

		}

			s_can_have_center_of_reformation_trigger = {
				RELIGION = s_lutheran
			}
		}
	}
	
	mean_time_to_happen = {
		months = 2500
		
		modifier = {
			factor = 0.4
			culture_group = midwestern_group
		}

		modifier = {
			factor = 0.50
			OR = {
				primary_culture = wisconsinite

				primary_culture = south_dakotan
				primary_culture = us_dakotan
				primary_culture = nebraskan
				primary_culture = iowan
			}
		}
				
		modifier = {
			factor = 0.75
			reform_desire = 1.0
		}
		modifier = {
			factor = 0.75
			reform_desire = 1.05
		}
		modifier = {
			factor = 0.75
			reform_desire = 1.10
		}
		modifier = {
			factor = 0.66
			reform_desire = 1.15
		}
		modifier = {
			factor = 0.30
			OR = {
				primary_culture = minnesotan
				primary_culture = north_dakotan
			}
		}
		modifier = {
			factor = 0.50
				reform_desire = 1.20
		}
		modifier = {
			factor = 0.25
			reform_desire = 1.25
		}
		
		modifier = {
			factor = 0.75
			NOT = { num_of_cities = 10 }
		}
		modifier = {
			factor = 0.75
			NOT = { num_of_cities = 2 }
		}
		
		modifier = {
			factor = 0.75
			has_idea_group = innovativeness_ideas
		}
		modifier = {
			factor = 1.25
			has_idea_group = religious_ideas
		}

			factor = 0.5
			num_of_rebel_controlled_provinces = 1
		}
		modifier = {
			factor = 2.0
			has_reform = feudal_theocracy

			factor = 10.0
			OR = {
				culture_group = southerner
				culture_group = caribbean_islander
				capital_scope = {
					superregion = central_america_superregion
				}
			}
		}
	}

		name = "superstates_greatawakening.1opta"
		enable_religion = s_lutheran

			limit = {
				s_can_have_center_of_reformation_trigger = {
