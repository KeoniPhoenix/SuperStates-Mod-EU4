		required_missions = { }
		position = 2

			NOT = { country_or_non_sovereign_subject_holds = ROOT }

			}

			}
			if = {
				limit = {
					primary_culture = north_carolinian
				}
				accepted_culture = south_carolinian
			}
			if = {
				limit = {
					primary_culture = south_carolinian
				}

				2523 = {
					country_or_non_sovereign_subject_holds = ROOT
